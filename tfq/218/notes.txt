3) Reproduce `ci-file-checks.yaml`
----------------------------------

The GitHub workflow only checks changed files.
These commands use:

```bash
git diff --name-only --diff-filter=AM origin/master...HEAD
```

If your base branch is not `origin/master`, replace it accordingly.


3.1 Python lint and format checks
---------------------------------

Use the same local Ubuntu 22.04 image:

```powershell
docker run --rm -it -v "C:\Users\Pablo Samano\Desktop\Rivian\google\quantum_p:/workspace/quantum" tfq-ci-build
```

Inside the container:

```bash
git config --global --add safe.directory /workspace/quantum
cd /workspace/quantum
sed -i 's/\r$//' configure.sh release/build_pip_package.sh scripts/*.sh || true

./scripts/ci_install.sh
git fetch origin master

CHANGED_PY="$(git diff --name-only --diff-filter=AM origin/master...HEAD -- '*.py')"
printf 'Changed Python files:\n%s\n' "${CHANGED_PY}"

if [[ -n "${CHANGED_PY}" ]]; then
  pylint -v ${CHANGED_PY}
  yapf --parallel --diff --style=google ${CHANGED_PY}
else
  echo "No changed Python files."
fi
```


3.2 C++ / Protobuf format check
-------------------------------

Use Ubuntu 24.04 to match the GitHub runner that provides `clang-format-18`:

```powershell
docker run --rm -it -v "C:\Users\Pablo Samano\Desktop\Rivian\google\quantum_p:/workspace/quantum" ubuntu:24.04 bash
```

Inside the container:

```bash
apt-get update
apt-get install -y git clang-format-18

git config --global --add safe.directory /workspace/quantum
cd /workspace/quantum
git fetch origin master

CHANGED_CC="$(git diff --name-only --diff-filter=AM origin/master...HEAD -- '*.cc' '*.h' '*.proto')"
printf 'Changed C++ / proto files:\n%s\n' "${CHANGED_CC}"

if [[ -n "${CHANGED_CC}" ]]; then
  clang-format-18 --verbose -Werror --style google --dry-run ${CHANGED_CC}
else
  echo "No changed C++ / proto files."
fi
```
